# TypeScript Types

Full type reference for `@classytic/media-kit`.

All types can be imported from the package root:

```typescript
import type { MediaKitConfig, IMediaDocument, StorageDriver, ... } from '@classytic/media-kit';
```

## Configuration Types

### MediaKitConfig

Main configuration object passed to `createMedia()`.

```typescript
interface MediaKitConfig {
  driver: StorageDriver;
  fileTypes?: FileTypesConfig;
  folders?: FolderConfig;
  processing?: ProcessingConfig;
  multiTenancy?: MultiTenancyConfig;
  deduplication?: DeduplicationConfig;
  softDelete?: SoftDeleteConfig;
  concurrency?: ConcurrencyConfig;
  logger?: MediaKitLogger;
  suppressWarnings?: boolean;
  plugins?: PluginType[];
  pagination?: PaginationConfig;
}
```

| Field | Type | Description |
|-------|------|-------------|
| `driver` | `StorageDriver` | Storage driver instance (required) |
| `fileTypes` | `FileTypesConfig` | File type restrictions |
| `folders` | `FolderConfig` | Folder configuration |
| `processing` | `ProcessingConfig` | Image processing config |
| `multiTenancy` | `MultiTenancyConfig` | Multi-tenancy config |
| `deduplication` | `DeduplicationConfig` | File deduplication config |
| `softDelete` | `SoftDeleteConfig` | Soft delete config |
| `concurrency` | `ConcurrencyConfig` | Concurrency control |
| `logger` | `MediaKitLogger` | Logger instance |
| `suppressWarnings` | `boolean` | Suppress warnings about missing optional dependencies |
| `plugins` | `PluginType[]` | Mongokit plugins (e.g., `cachePlugin`) |
| `pagination` | `PaginationConfig` | Mongokit pagination config |

## Storage Driver Types

### StorageDriver

The core interface for storage backends. Implement this to create custom drivers.

```typescript
interface StorageDriver {
  readonly name: string;

  // Core (required)
  write(key: string, data: Buffer | NodeJS.ReadableStream, contentType: string): Promise<WriteResult>;
  read(key: string, range?: { start: number; end: number }): Promise<NodeJS.ReadableStream>;
  delete(key: string): Promise<boolean>;
  exists(key: string): Promise<boolean>;
  stat(key: string): Promise<FileStat>;

  // Navigation (optional)
  list?(prefix: string): AsyncIterable<string>;
  copy?(source: string, destination: string): Promise<WriteResult>;
  move?(source: string, destination: string): Promise<WriteResult>;

  // URLs
  getPublicUrl(key: string): string;
  getSignedUrl?(key: string, expiresIn?: number): Promise<string>;
  getSignedUploadUrl?(key: string, contentType: string, expiresIn?: number): Promise<PresignedUploadResult>;
}
```

---

| Field | Type | Description |

### PresignedUploadResult

Returned from `StorageDriver.getSignedUploadUrl()` and `media.getSignedUploadUrl()`.

```typescript
interface PresignedUploadResult {
  uploadUrl: string;
  key: string;
  publicUrl: string;

  headers?: Record<string, string>;
}
```

| Field | Type | Description |
|-------|------|-------------|
| `uploadUrl` | `string` | The presigned URL to PUT the file to |
| `key` | `string` | Storage key assigned to the file |
| `publicUrl` | `string` | Public URL where the file will be accessible after upload |
| `expiresIn` | `number` | Time in seconds until the presigned URL expires |
| `headers` | `Record<string, string>` | Optional headers the client must include with the PUT request |

## Document Types

### IMedia

Base interface for media documents.

```typescript
interface IMedia {
  // Identity (Directus triple-name pattern)
  filename: string;
  originalFilename: string;
  title: string;

  // Storage
  key: string;
  url: string;
  mimeType: string;
  size: number;
  hash: string;

  // Status lifecycle
  status: MediaStatus;
  errorMessage?: string;

  // Organization
  folder: string;
  tags: string[];
  alt?: string;
  description?: string;

  // Image metadata
  width?: number;
  height?: number;
  aspectRatio?: number;
  focalPoint?: FocalPoint;

  // Variants
  variants: GeneratedVariant[];

  // Video/Audio
  duration?: number;

  // Extensible
  metadata: Record<string, unknown>;
  exif?: ExifMetadata;

  // Soft delete
  deletedAt?: Date | null;

  // Audit
  uploadedBy?: Types.ObjectId;
  organizationId?: Types.ObjectId | string;

  // Timestamps
  createdAt: Date;
  updatedAt: Date;
}
```

### MediaStatus

File status lifecycle.

```typescript
type MediaStatus = 'pending' | 'processing' | 'ready' | 'error';
```

The lifecycle is: `pending` (record created) -> `processing` (being processed/uploaded) -> `ready` (success) or `error` (failure).

---

### FocalPoint

Focal point for smart cropping. Inspired by Payload CMS.

| Field | Type | Range | Description |
|-------|------|-------|-------------|
| `x` | `number` | `0.0` - `1.0` | Horizontal position. `0.0` = left, `1.0` = right |
| `y` | `number` | `0.0` - `1.0` | Vertical position. `0.0` = top, `1.0` = bottom |

## Processing Types

### SizeVariant

Configuration for a size variant to generate during upload.

```typescript
interface SizeVariant {
  name: string;
  width?: number;
  height?: number;
  aspectRatio?: AspectRatioPreset;
  quality?: number;
  format?: 'webp' | 'jpeg' | 'png' | 'avif' | 'original';
  condition?: (original: { width: number; height: number; mimeType: string }) => boolean;
}
```

| Field | Type | Description |
|-------|------|-------------|
| `name` | `string` | Variant name (e.g., `'thumbnail'`, `'medium'`, `'large'`) |
| `width` | `number` | Maximum width in pixels |
| `height` | `number` | Maximum height in pixels |
| `aspectRatio` | `AspectRatioPreset` | Aspect ratio preset |
| `quality` | `number` | Output quality (1-100) |
| `format` | `string` | Output format (defaults to processing config format) |
| `condition` | `function` | Skip variant if returns `false` (Payload pattern) |

The `condition` function allows conditional variant generation based on original image properties.

---

### GeneratedVariant

Stored per-variant in the media document. Each variant has independent `mimeType` and `filename`.

```typescript
interface GeneratedVariant {
  name: string;
  key: string;
  url: string;
  filename: string;
  mimeType: string;
  size: number;
  width?: number;
  height?: number;
}
```

Variants store their own `mimeType` and `filename` (Payload pattern). A JPEG original can have WebP thumbnails.

---

## Event Types

### MediaEventName

All event names supported by the event system. Events follow the pattern `phase:operation`.

```typescript
type MediaEventName =
  // Upload
  | 'before:upload' | 'after:upload' | 'error:upload'
  | 'before:uploadMany' | 'after:uploadMany' | 'error:uploadMany'
  // Delete
  | 'before:delete' | 'after:delete' | 'error:delete'
  | 'before:deleteMany' | 'after:deleteMany' | 'error:deleteMany'
  // Move
  | 'before:move' | 'after:move' | 'error:move'
  // Replace
  | 'before:replace' | 'after:replace' | 'error:replace'
  // Soft delete
  | 'before:softDelete' | 'after:softDelete'
  // Restore
  | 'before:restore' | 'after:restore'
  // Import from URL
  | 'before:import' | 'after:import' | 'error:import'
  // Presigned uploads
  | 'before:presignedUpload' | 'after:presignedUpload'
  | 'before:confirmUpload' | 'after:confirmUpload'
  // Folder operations
  | 'before:rename' | 'after:rename'
  // Processing
  | 'before:validate' | 'after:process';
```

---

### EventListener

Event listener function. Can be synchronous or async -- async listeners are awaited.

```typescript
type EventListener<T = unknown> = (payload: T) => void | Promise<void>;
```

Listeners can return `Promise<void>` and will be awaited.

---

### Unsubscribe

Returned by `on()`. Call to remove the listener.

```typescript
type Unsubscribe = () => void;
```

`on()` returns an unsubscribe function instead of `void`.

---

### EventResult

## MediaKit Instance Type

### MediaKit

The main instance interface returned by `createMedia()`.

```typescript
interface MediaKit {
  readonly config: MediaKitConfig;
  readonly driver: StorageDriver;
  readonly schema: Schema<IMediaDocument>;
  readonly repository: MediaRepository;

  init(model: MediaModel): this;

  // Events
  on<T = unknown>(event: MediaEventName, listener: EventListener<T>): Unsubscribe;
  emit<T = unknown>(event: MediaEventName, payload: T): Promise<void>;

  // Core operations
  upload(input: UploadInput, context?: OperationContext): Promise<IMediaDocument>;
  uploadMany(inputs: UploadInput[], context?: OperationContext): Promise<IMediaDocument[]>;
  delete(id: string, context?: OperationContext): Promise<boolean>;
  deleteMany(ids: string[], context?: OperationContext): Promise<BulkResult>;
  move(ids: string[], targetFolder: string, context?: OperationContext): Promise<{ modifiedCount: number }>;

  // File replacement
  replace(id: string, input: UploadInput, context?: OperationContext): Promise<IMediaDocument>;

  // Soft deletes
  softDelete(id: string, context?: OperationContext): Promise<IMediaDocument>;
  restore(id: string, context?: OperationContext): Promise<IMediaDocument>;
  purgeDeleted(olderThan?: Date, context?: OperationContext): Promise<number>;

  // URL import
  importFromUrl(url: string, options?: ImportOptions, context?: OperationContext): Promise<IMediaDocument>;

  // Tags
  addTags(id: string, tags: string[], context?: OperationContext): Promise<IMediaDocument>;
  removeTags(id: string, tags: string[], context?: OperationContext): Promise<IMediaDocument>;

  // Focal point
  setFocalPoint(id: string, focalPoint: FocalPoint, context?: OperationContext): Promise<IMediaDocument>;

  // Queries
  getById(id: string, context?: OperationContext): Promise<IMediaDocument | null>;
  getAll(params?: { ... }, context?: OperationContext): Promise<OffsetPaginationResult | KeysetPaginationResult>;
  search(query: string, params?: { ... }, context?: OperationContext): Promise<OffsetPaginationResult | KeysetPaginationResult>;

  // Presigned uploads
  getSignedUploadUrl(filename: string, contentType: string, options?: { ... }): Promise<PresignedUploadResult>;
  confirmUpload(input: ConfirmUploadInput, context?: OperationContext): Promise<IMediaDocument>;

  // Folder operations
  getFolderTree(context?: OperationContext): Promise<FolderTree>;
  getFolderStats(folder: string, context?: OperationContext): Promise<FolderStats>;
  getBreadcrumb(folder: string): BreadcrumbItem[];
  deleteFolder(folder: string, context?: OperationContext): Promise<BulkResult>;
  renameFolder(oldPath: string, newPath: string, context?: OperationContext): Promise<{ modifiedCount: number }>;
  getSubfolders(parentPath: string, context?: OperationContext): Promise<FolderNode[]>;

  // Utilities
  validateFile(buffer: Buffer, filename: string, mimeType: string): void;
  getContentType(folder: string): string;
}
```

## Usage Example

```typescript
import { createMedia } from '@classytic/media-kit';
import { S3Provider } from '@classytic/media-kit/providers/s3';
import type {
  MediaKitConfig,
  IMediaDocument,
  UploadInput,
  OperationContext,
  StorageDriver,
  FocalPoint,
} from '@classytic/media-kit';

const config: MediaKitConfig = {
  driver: new S3Provider({
    bucket: 'my-uploads',
    region: 'us-east-1',
  }),
  folders: {
    defaultFolder: 'general',
  },
  deduplication: {
    enabled: true,
    algorithm: 'sha256',
  },
  softDelete: {
    enabled: true,
    ttlDays: 30,
  },
};

const media = createMedia(config);

const input: UploadInput = {
  buffer: fileBuffer,
  filename: 'photo.jpg',
  mimeType: 'image/jpeg',
  tags: ['product', 'hero'],
  description: 'Main hero image',
  focalPoint: { x: 0.5, y: 0.3 },
};

const ctx: OperationContext = {
  userId: 'user_123',
  organizationId: 'org_456',
};

const file: IMediaDocument = await media.upload(input, ctx);
```
