# Quick Start

Get up and running with `@classytic/media-kit` in 5 minutes.

## Step 1: Connect to MongoDB

```typescript
import mongoose from 'mongoose';

await mongoose.connect('mongodb://localhost:27017/myapp');
```

## Step 2: Create a Storage Driver

### AWS S3

```bash
npm install @aws-sdk/client-s3
```

```typescript
import { S3Provider } from '@classytic/media-kit/providers/s3';

const driver = new S3Provider({
  bucket: 'my-bucket',
  region: 'us-east-1',
  credentials: {
    accessKeyId: process.env.AWS_ACCESS_KEY_ID,
    secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY,
  },
});
```

### Google Cloud Storage

```bash
npm install @google-cloud/storage
```

```typescript
import { GCSProvider } from '@classytic/media-kit/providers/gcs';

const driver = new GCSProvider({
  bucket: 'my-bucket',
  projectId: 'my-project',
  keyFilename: './service-account.json',
});
```

## Step 3: Initialize Media Kit

```typescript
import { createMedia } from '@classytic/media-kit';

const media = createMedia({ driver });

// Create Mongoose model
const Media = mongoose.model('Media', media.schema);

// Initialize with the model
media.init(Media);
```

## Step 4: Upload a File

```typescript
// From Express/Multer
app.post('/upload', upload.single('file'), async (req, res) => {
  const uploaded = await media.upload({
    buffer: req.file.buffer,
    filename: req.file.originalname,
    mimeType: req.file.mimetype,
    tags: ['user-upload'],
    description: 'Uploaded via API',
  });

  res.json({
    url: uploaded.url,
    status: uploaded.status, // 'ready'
  });
});
```

## Complete Example

```typescript
import express from 'express';
import mongoose from 'mongoose';
import multer from 'multer';
import { createMedia } from '@classytic/media-kit';
import { S3Provider } from '@classytic/media-kit/providers/s3';

// Setup
await mongoose.connect('mongodb://localhost:27017/myapp');

const media = createMedia({
  driver: new S3Provider({
    bucket: process.env.S3_BUCKET,
    region: 'us-east-1',
    credentials: {
      accessKeyId: process.env.AWS_ACCESS_KEY_ID,
      secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY,
    },
  }),
});

const Media = mongoose.model('Media', media.schema);
media.init(Media);

// Express route
const app = express();
const upload = multer();

app.post('/upload', upload.single('file'), async (req, res) => {
  const file = await media.upload({
    buffer: req.file.buffer,
    filename: req.file.originalname,
    mimeType: req.file.mimetype,
    folder: 'images',
    tags: ['product'],
    description: 'Product image',
  });

  res.json({
    id: file._id,
    url: file.url,
    status: file.status,
    hash: file.hash,
    tags: file.tags,
  });
});

app.listen(3000);
```
