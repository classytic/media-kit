# Basic Upload Example

Simple file upload with Express and Multer.

## Setup

```bash
npm install express multer @classytic/media-kit @classytic/mongokit mongoose @aws-sdk/client-s3
```

## Code

```typescript
import express from 'express';
import multer from 'multer';
import mongoose from 'mongoose';
import { createMedia } from '@classytic/media-kit';
import { S3Provider } from '@classytic/media-kit/providers/s3';

// Connect to MongoDB
await mongoose.connect('mongodb://localhost:27017/myapp');

// Create media kit with S3 driver
const media = createMedia({
  driver: new S3Provider({
    bucket: process.env.S3_BUCKET!,
    region: process.env.S3_REGION!,
    credentials: {
      accessKeyId: process.env.AWS_ACCESS_KEY_ID!,
      secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY!,
    },
  }),
});

const Media = mongoose.model('Media', media.schema);
media.init(Media);

// Express app
const app = express();
const upload = multer();

// Upload endpoint
app.post('/upload', upload.single('file'), async (req, res) => {
  try {
    const file = await media.upload({
      buffer: req.file.buffer,
      filename: req.file.originalname,
      mimeType: req.file.mimetype,
      folder: 'images',
      tags: ['upload'],
    });

    res.json({
      success: true,
      file: {
        id: file._id,
        url: file.url,
        filename: file.originalFilename,
        size: file.size,
        status: file.status,
      },
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message,
    });
  }
});

// List files
app.get('/files', async (req, res) => {
  const files = await media.getAll({
    sort: { createdAt: -1 },
    limit: 20,
  });

  res.json(files);
});

// Delete file
app.delete('/files/:id', async (req, res) => {
  const deleted = await media.delete(req.params.id);
  res.json({ success: deleted });
});

// Soft delete (trash)
app.post('/files/:id/trash', async (req, res) => {
  const file = await media.softDelete(req.params.id);
  res.json({ success: true, deletedAt: file.deletedAt });
});

// Restore from trash
app.post('/files/:id/restore', async (req, res) => {
  const file = await media.restore(req.params.id);
  res.json({ success: true, file });
});

app.listen(3000, () => {
  console.log('Server running on http://localhost:3000');
});
```

## Test with cURL

```bash
# Upload file
curl -F "file=@photo.jpg" http://localhost:3000/upload

# List files
curl http://localhost:3000/files

# Delete file
curl -X DELETE http://localhost:3000/files/FILE_ID

# Soft delete
curl -X POST http://localhost:3000/files/FILE_ID/trash

# Restore
curl -X POST http://localhost:3000/files/FILE_ID/restore
```
